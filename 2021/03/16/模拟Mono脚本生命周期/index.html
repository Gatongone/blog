模拟Mono脚本生命周期

为了加深对Mono生命周期和GameObject的印象，特地尝试了模拟Mono工作原理。

分析思路如下：

  1. unity是如何组织场景中的对象的————————————场景管理每一个gameobject,gameobject挂载各个脚本

  2. 继承了mono的脚本可以实现start、update等生命周期的函数——————————————mono基类应该以virtual声明的生命周期函数

  3. mono是挂载在gameobject身上的组件，并且可以挂载多个————————————gameobject中应该存在一个mono容器

  4. 可以从挂载的脚本获得被挂载的gameobject——————————————mono中应该有gameobject字段

  5. 每个脚本的每个生命周期似乎都是同时执行的——————————————unity中应该有并发处理每个生命周期函数的方法

  6. unity不支持多线程————————————那么或许有一个专门负责执行每个脚本的生命周期的大mono

综上可大概获得几个关键类：
* GameObject：负责管理Mono，并将所管理的mono发布到MainMono上
* MonoBehaviour：一个基类，有各种各样的生命周期函数和一个gameobject字段
* MainMono：接收能进入生命周期脚本的mono，并按生命周期顺序执行他们的生命周期

关键类

  • MonoBehaviour

    为子类提供生命周期函数，需要通知gameobject哪些脚本被启用或销毁了，以此通知MainMono禁用或启用该脚本的生命周期函数

    public class MonoBehaviour
{
    public bool IsEnabled = true;
    public GameObject gameobject;
    public virtual void Awake(){}
    public virtual void Start(){}
    public virtual void Update(){}
    public virtual void OnEnable(){}
    public virtual void OnDisable(){}
    public virtual void OnDestroy(){}

    //销毁脚本
    public void Destroy()
    {
        OnDestroy();
        this.gameobject.DestroyComponent(this);
    }
    //不启用脚本
    public void  Disable()
    {
        OnDisable();
        this.gameobject.DisableComponent(this);
    }
    //启用脚本
    public void  Enable()
    {
        OnEnable();
        this.gameobject.EnableComponent(this);
    }
}

  • MainMono

    通过单例委托接收MainMono发过来的脚本生命周期函数

    public class MainMono
{
    public static readonly Lazy<MainMono> instance = new Lazy<MainMono>(() => new MainMono());
    public static MainMono Instance { get => instance.Value; }

    public Action Awake;
    public Action OnEnable;
    public Action Start;
    public Action Update;

    //模拟游戏开始
    public void GameStart()
    {
         Console.WriteLine("游戏开始！");

        if(Awake!=null)
            Awake.Invoke();
        if(OnEnable!=null)
            OnEnable.Invoke();
        if(Start!=null)
            Start.Invoke();
        
        while (true)
        {
            if(Update!=null)
                Update.Invoke();

            Thread.Sleep(1000);
        }
    }
}

  • GameObject

    以列表存储mono脚本，通过可变长参数在初始化时将脚本加入列表并发布给MainMono

    public class GameObject
{
    public string Name;

    private List<MonoBehaviour> componentlist = new List<MonoBehaviour>();

    public Action OnDestroy;
    public Action OnDisable;

    public GameObject(params MonoBehaviour[] args)
    {
        foreach (var component in args)
        {
            //加入组件列表
            componentlist.Add(component);

            //设置gameobjct
            component.gameobject = this;

            //初始化到主Mono
            MainMono.Instance.Awake += component.Awake;
            MainMono.Instance.OnEnable += component.OnEnable;
            MainMono.Instance.Start += component.Start;
            MainMono.Instance.Update += component.Update;
            this.OnDestroy += component.OnDestroy;

            //如果不启用则不加入
            if (!component.IsEnabled)
            {
                MainMono.Instance.OnEnable -= component.OnEnable;
                MainMono.Instance.Start -= component.Start;
                MainMono.Instance.Update -= component.Update;
            }
        }
    }
    //添加脚本
    public void AddComponent<T>(T component)where T : MonoBehaviour
    {
        componentlist.Add(component);
        component.gameobject = this;

        //初始化到主Mono
        MainMono.Instance.Awake += component.Awake;
        MainMono.Instance.OnEnable += component.OnEnable;
        MainMono.Instance.Start += component.Start;
        MainMono.Instance.Update += component.Update;
        this.OnDestroy += component.OnDestroy;
        //如果不启用则不加入
        if (!component.IsEnabled)
        {
            MainMono.Instance.OnEnable -= component.OnEnable;
            MainMono.Instance.Start -= component.Start;
            MainMono.Instance.Update -= component.Update;
        }
    }
    //获取脚本
    public T GetComponent<T>()where T : MonoBehaviour
    {
       foreach (var item in componentlist)
       {
           if(item is T)
                return item as T;
       }
        return null;
    }
    //移除脚本
    public void DestroyComponent(MonoBehaviour component)
    {            
        //---------
        //移除逻辑
        //---------
        componentlist.Remove(component);

        MainMono.Instance.Awake -= component.Awake;
        MainMono.Instance.OnEnable -= component.OnEnable;
        MainMono.Instance.Start -= component.Start;
        MainMono.Instance.Update -= component.Update;
        this.OnDestroy -= component.OnDestroy;
    }
    //禁用脚本
    public void DisableComponent(MonoBehaviour component)
    {
        MainMono.Instance.OnEnable -= component.OnEnable;
        MainMono.Instance.Start -= component.Start;
        MainMono.Instance.Update -= component.Update;
    }
    //启用脚本
    public void EnableComponent(MonoBehaviour component)
    {
        MainMono.Instance.OnEnable += component.OnEnable;
        MainMono.Instance.Start += component.Start;
        MainMono.Instance.Update += component.Update;
    }
}

public class LogComponent_1:MonoBehaviour
{
    int i = 1;
    public override void Awake()=>Console.WriteLine("Logger1 Awake");
    public override void Start()=> Console.WriteLine("Logger1 Start");
    public override void Update()
    {
       Console.WriteLine("Logger1 Update");
       
        if (i==3)
        {
            //模拟3帧后执行销毁并添加一个脚本
            this.gameobject.AddComponent<LogComponent_2>(new LogComponent_2());
            this.Destroy();
        }
        i++;     
    }

    public override void OnDisable()=>Console.WriteLine("Logger1 Disable");
    public override void OnDestroy()=>Console.WriteLine("Logger1 Destroy");
