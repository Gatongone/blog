Persistence——XML

XML(Extensible Markup Language)即可拓展标记语言，作为互联网传输工具，无关于开发平台，可以实现不同系统之间的数据交互。(Extensible Markup Language)

  • 相关操作

    • 加载xml文件

      一般有三种方法：

      • XmlDocument：把数据加载到内存中进行读取
      • XmlTextReader：通过流形式读取，能减少内存占用，但是是单向只读的
      • LINQ to XML：一种启用了 LINQ 的内存 XML 编程接口
        //使用XmlDocument加载
XmlDocument xml = new XmlDocument();
//方法一：根据xml文件名，通过加载文本文件再翻译成xml对象
TextAsset asset = Resources.Load<TextAsset>("TextXml.xml");
xml.LoadXml(asset.text);
//方法二：根据xml路径直接加载
xml.Load(Application.persistentDataPath + "TextXml.xml" );

    • 读取xml信息

      <?xml version="1.0" encoding="UTF-8"?>
<Root>
    <name>关东关</name>
    <item id="1" num="20"/>
    <list>
        <name>小明</name>
        <age>18</age>
    </list>
    <list>
        <name>小红</name>
        <age>17</age>
    </list>
</Root>

      //获取节点中的信息：
//获取根节点
XmlNode root = xml.SelectSingleNode("Root");
//再获取子节点
xmlNode nodeName = root.SelectSingleNode("name");
//获取子节点中的信息
string name = nodeName.InnerText as string;

//获取结点中属性的信息：
xmlNode nodeItem = root.SelectSingleNode("item");
//获取属性
//方法一：
string id = nodeItem.Attributes["id"].Value as string;
int num = int.Parse(nodeItem.Attributes["num"].Value);
//方法二：
id = nodeItem.Attributes.GetNamedItem("id").Value as string;
num = int.Parse(nodeItem.Attributes.GetNamedItem("num").Value);

//获取某结点下的所有同名结点
XmlNodeList list = root.SelectNodes("list");
//遍历获取信息
foreach (XmlNode item in list)
{
    Debug.Log(item.SelectSingleNode("name").InnerText);
    Debug.Log(item.SelectSingleNode("age").InnerText);   
}

    • 存储xml文件

      //1.创建xml文档
XmlDocument xml = new XmlDocument();

//2.添加版本声明
//创建版本声明
//参数：版本号、编码类型、是否依赖于外部定义填"yes"或"no"(可以不填)
XmlDeclaration xmlDec = xml.CreateXmlDeclaration("1.0", "UTF-8", "");
//添加版本声明到xml文档中
xml.AppendChild(xmlDec);

//3.添加一个根节点
XmlElement root = xml.CreateElement("root");
xml.AppendChild(root);

//4.在根节点下添加一个子节点
XmlElement node = xml.CreateElement("node");
//为节点添加信息
node.InnerText = "Test info";
//加入到根节点下
root.AppendChild(node);

//4.5.添加属性
XmlElement attribute = xml.CreateElement("attribute");
attribute.SetAttribute("id", 1.ToString());
attribute.SetAttribute("value", "属性的值");
root.AppendChild(attribute);

//5.保存xml到路径
xml.Save(Application.persistentDataPath + "/Testxml.xml");

      生成的xml：

      <?xml version="1.0" encoding="UTF-8"?>
<root>
    <node>Test info</node>
    <attribute id="1" value="属性的值" />
</root>

    • 修改xml文件

      //1.判断文件是否存在
if (File.Exists(path))
{
    //2.加载xml文档
    XmlDocument xml = new XmlDocument();
    xml.Load(path);

    XmlNode root = xml.SelectSingleNode("root");
    //3.移除一条节点:找到子节点的父节点再移除
    XmlNode node = xml.SelectSingleNode("root/node");
    root.RemoveChild(node);

    //4.添加一条节点
    XmlElement newnode = xml.CreateElement("newnode");
    newnode.InnerText = "一个新的节点";
    root.AppendChild(newnode);

    //5.保存xml到路径
    xml.Save(path);
}

      修改后的xml

      <?xml version="1.0" encoding="UTF-8"?>
<root>
    <attribute id="1" value="属性的值" />
    <newnode>一个新的节点</newnode>
</root>

  • Xml的序列化和反序列化

    • 序列化

      通过序列化的方式将对象实例直接翻译成xml文件,相当于为实例对象的字段和属性序列化（若字段或属性已实现了IXmlSerializable接口，则调用字段的WriteXml方法）

      注意事项：

      1. 这种方式不能序列化字典
      2. 如果序列化的类型用了有参的构造函数，该类型必须拥有一个公有的无参构造函数
      3. 并且空数据将不会生成节点，比如下面的testnull
      4. 只能对public的数据进行序列化

      测试对象

      public class SerializeTest
{
    public int testPublic = 1;
    private int testPrivate = 2;
    protected int testProtected = 3;
    internal int testInternal = 4;

    public string testStr = "testStr";
    public string testnull;
    public int testPro{ get; set; }
    public TestInfo testinfo = new TestInfo(0,"测试信息");

    //这里只是为了演示
    //强烈建议不要在类内对list和array进行初始化
    //否则在反序列化时将会在初始化后的list和array的基础上，再次加入重复数据
    public List<TestInfo> infolist = new List<TestInfo>()
    {
        new TestInfo(1,"列表1"),
        new TestInfo(2,"列表2"),
        new TestInfo(3,"列表3"),
    };

}
public class TestInfo
{
    public TestInfo(int id, string info) => (this.id, this.info) = (id, info);
    public TestInfo(){}//必须有一个无参的构造函数

    [XmlAttribute()]
    public int id;
    [XmlAttribute()]
    public string info;
}

      序列化过程

          //序列化测试对象
    SerializeTest s1 = new SerializeTest();
    //存储路径
    string path = Application.persistentDataPath + "/SerializableXml.xml";
    
    //1.对xml文件 创建 写入流
    //StreamWriter(string path)：
    //如果路径文件存在，则为文件创建一个写入流
    //如果路径文件不存在则创建文件，并为其创建一个写入流
    using( StreamWriter writer = new StreamWriter(path))
    {
        //2.声明一个 SerializeTest类型 的序列化器,用于将该类型编码成xml格式
        XmlSerializer serializer = new XmlSerializer(typeof(SerializeTest));
        //3.序列化器 序列化 实例对象
        //先将 s1 序列化后加入到 写入流中
        //再调用写入流中的Write
        serializer.Serialize(writer, s1);
    }

[XmlAttribute()]            //用于将 字段 在xml中标记成 属性

    • 反序列化

      通过反序列化将读取xml文件并赋值给类对象实例，相当于实例中的公有字段和属性均反序列化（若字段或属性已实现了IXmlSerializable接口，则调用字段的ReadXml方法）

          //读取路径
    string path = Application.persistentDataPath + "/SerializableXml.xml";

    //1.判断路径是否存在
    if(File.Exists(path))
    {
        //2.创建读取流
        using(StreamReader reader = new StreamReader(path))
        {
            //3.创建序列化器
            XmlSerializer serializer = new XmlSerializer(typeof(SerializeTest));
            //4.反序列化生成实例对象
            SerializeTest test = serializer.Deserialize(reader) as SerializeTest;
        }
    }

  • IXmlSerializable接口

    由于XmlSerializer对于某些对象不能进行处理（比如Dictionary和空数据），因此需要实现IXmlSerializable接口来对这些对象来进行自定义的处理
返回值 接口成员方法 描述
XmlSchema GetSchema() 保留的方法，不应该被使用，应从此方法返回一个null
void ReadXml(XmlReader reader) 反序列化（XmlSerializer.Deserialize()）时调用该方法
void WriteXml(XmlWriter writer) 序列化（XmlSerializer.Serialize()）时调用该方法

  • 接口成员方法的使用

    序列化后的xml：

    <?xml version="1.0" encoding="utf-8"?>
<SerializeTest>
    <strNode>节点字符串</strNode>
    <numNode>7385</numNode>
    <Attribute strAttribute="属性字符串" numAttribute="2333" />
    <childNode>
        <int>5</int>
    </childNode>
</SerializeTest>

    实现IXmlSerializable接口的类对象

    public class SerializeTest : IXmlSerializable
{
    public string strNode;
    public int numNode;
    public string strAttribute;
    public int numAttribute;
    public string childNode;

    //返回结构
    public XmlSchema GetSchema()
    {
        return null;
    }

    //用于编写自定义反序列化规则
    public void ReadXml(XmlReader reader)
    {
        //读节点
        //方法一:
        reader.Read();//读到 节点<strNode>  类型为 XmlNodeType.Element
        reader.Read();//读到 "节点字符串"    类型为 XmlNodeType.Text
        this.strNode = reader.Value; 
        reader.Read();//读到 节点</strNode> 类型为 XmlNodeType.EndElement
        reader.Read();//读到 节点<numNode>  类型为 XmlNodeType.Element
        reader.Read();//读到 7385           类型为 XmlNodeType.Text
        this.numNode = int.Parse(reader.Value);
        //方法二
        while (reader.Read())
        {
            if (reader.NodeType == XmlNodeType.Element)
            {
                switch (reader.Name)
                {
                    case "strNode":
                        reader.Read();//读下一个
                        this.strNode = reader.Value;
                        break;
                    case "num":
                        reader.Read();//读下一个
                        this.numNode = int.Parse(reader.Value);
                        break;
                }
            }
        }

        //读属性
        this.strAttribute = reader["strAttribute"];
        this.numAttribute = int.Parse(reader["numAttribute"]);

        //读子节点
        XmlSerializer serializer = new XmlSerializer(typeof(int));
        reader.Read();
        reader.ReadStartElement("childNode");
        serializer.Deserialize(reader);
        reader.ReadEndElement();
    }

    //用于编写自定义序列化规则
    public void WriteXml(XmlWriter writer)
    {
        //写节点
        writer.WriteElementString("strNode", "节点字符串");
        writer.WriteElementString("numNode", 7385.ToString());

        //写属性
        writer.WriteStartElement("Attribute");
        writer.WriteAttributeString("strAttribute", "属性字符串");
        writer.WriteAttributeString("numAttribute", 2333.ToString());
        writer.WriteEndElement();
        
        //写子节点
        XmlSerializer serializer = new XmlSerializer(typeof(int));
        writer.WriteStartElement("childNode");
        serializer.Serialize(writer, 5);
        writer.WriteEndElement();
    }
}

    • 示例：自定义序列化字典

      public class SerializableDictionary<TKey, TValue> : Dictionary<TKey, TValue>, IXmlSerializable
{
  public XmlSchema GetSchema()
  {
      return null;
  }

  //反序列化时
  public void ReadXml(XmlReader reader)
  {
      XmlSerializer keySerializer = new XmlSerializer(typeof(TKey));
      XmlSerializer valueSerializer = new XmlSerializer(typeof(TValue));

      //跳过父节点
      reader.Read();

      //如果不是</>节点则持续反序列化
      while(reader.NodeType!=XmlNodeType.EndElement)
      {
          //反序列化元素后reader将跳过该元素
          //比如
          //<node>
          //  <key>测试</key>
          //  <value>5</value>
          //</node>
          //反序列化key后，下一次将直接读取<value>,而不会读取到</key>
          //也就是说子元素的</>不会影响到反序列化
          keySerializer.Deserialize(reader);
          valueSerializer.Deserialize(reader);
      }
  }

  //序列化时
  public void WriteXml(XmlWriter writer)
  {
      XmlSerializer keySerializer = new XmlSerializer(typeof(TKey));
      XmlSerializer valueSerializer = new XmlSerializer(typeof(TValue));
      //序列化时遍历本类字典
      foreach(KeyValuePair<TKey, TValue> kv in this)
      {
          //分别对键值序列化
          keySerializer.Serialize(writer,kv.Key);
          valueSerializer.Serialize(writer, kv.Value);
      }
  }
}
