XLua Hotfix方案

  • 使用方式

    1.添加HOTFIX_ENABLE宏

    如果是第一次使用Xlua做Hotfix，需要在Unity3D的File->Build Setting->Scripting Define Symbols下添加 HOTFIX_ENABLE 宏。

    设置宏后会在Xlua菜单下生成 Hotfix Inject In Editor 选项

    注意，该宏必须在编辑器下设置，在代码里头设置不会生效。编辑器、各手机平台这个宏要分别设置。

    2.为需要做Hotfix的类加上[Hotfix]特性

    通过该特性，xlua可以将要注入的代码加到Hotfix列表

    3.生成代码

    执行 XLua/Generate Code 菜单，生成的代码可以在 Xlua\Gen 看到

    注意，每次对Hotfix列表中的类新增方法，都要重新生成一次代码

    4.注入代码到Hotfix列表

    手动执行 XLua/Hotfix Inject In Editor 菜单，注入完成后会打印 hotfix inject finish! 或者 had injected! ，否则打印错误信息

    若打印 please install the Tools 说明缺少Tools依赖，需要将Tools文件夹导入到与Assets同级的目录下

    注意，每次修改了Hotfix列表中的类，都要重新执行一次注入

  • 在Lua中做C#的热补丁

    Cshap部分，这里举例了最常用的几个需要打热补丁的代码，这里的Log以及Lua中的log都是我自己封装的控制台输出方法，就当成 Debug.Log 和 print 就行

    public class Slime{}
public class Goblin{}

[Hotfix]
public class Monster<T>
{
    public void PrintMonsterType()
    {
        this.Log("Monster");
    }
}

[Hotfix]
public class DataInfo
{
    int[] array;
    public string Name
    {
        get => "Cshap中的属性";
        set {}
    }
    public int this[int index]
    {
        get
        {
            if(array.Length <= index || index<0)
                throw new IndexOutOfRangeException();
            return array[index];
        }
        set
        {
            if(index<0)
                throw new IndexOutOfRangeException();
            array[index] = value;
        }
    }
    public DataInfo() => this.Log("这里是Cshap中的构造函数");
    ~DataInfo() => Debug.Log("这里是Cshap中的析构函数");
    public void Print()
    {
        this.Log("这里是Cshap中的成员方法");
    }
    public IEnumerator Timer()
    {
        while (true)
        {
            this.Log("打印一次Cshap中的协程");
            yield return new WaitForSeconds(1f);
        }
    }

}

[Hotfix]
public class HotfixTest : MonoBehaviour
{
    void Start()
    {
        //自己封装的初始化
        LuaMgr.Instance.Init();
        LuaMgr.Instance.Require("main");

        var datainfo = new DataInfo();

        datainfo.Name = "修改后的姓名";
        this.Log(datainfo.Name);

        datainfo[0] = 2;
        Debug.Log(datainfo[0]);

        datainfo.Print();
        StartCoroutine(datainfo.Timer());

        var slime = new Monster<Slime>();
        var goblin = new Monster<Goblin>();
        slime.PrintMonsterType();
        goblin.PrintMonsterType();
    }
    void Add(int a, int b)
    {
        this.Log("我猜不会打印出来");
    }
}

    Lua部分，对Cshap打热补丁，注意对比输出的内容

    --导入工具包，下面重写协程要用
local util = require "xlua.util"

--可以这样，用名字做引用，但只能写一个

xlua.hotfix(CS.HotfixTest,"Add",function (a,b)
    log("执行加法运算："..a+b)
end)

--也可以合一块儿写

xlua.hotfix(CS.DataInfo,{

    --重写属性
    get_Name = function (self)
        return "Lua中get属性"
    end,
    set_Name = function (self,value)
        log("Lua中set属性："..value)
    end,

    --重写索引器,固定写法
    get_Item = function (self,index)
        log("尝试在lua中重写get索引器")
    end,
    set_Item = function (self,index,value)
        log("尝试在lua中重写set索引器,传入值",value)
    end,

    --重写协程
    Timer = function ()
        --需要返回一个协程
        return util.cs_generator(function ()
            i = 1
            while i<3 do
                log("打印一次lua中的协程")
                coroutine.yield( CS.UnityEngine.WaitForSeconds(1) )
                i = i + 1
            end
        end)
    end,

    --重写成员方法
    Print = function ()
        log("这是在Lua中打了Hotfix后的成员方法")
    end,

    --构造函数固定写法
    [".ctor"] = function ()
        log("这是在Lua中打了Hotfix后的构造函数")
    end,

    --析构函数固定写法
    Finalize = function ()
        log("这是在Lua中打了Hotfix后的析构函数")
    end
})

--泛型类替换

xlua.hotfix(CS.Monster(CS.Slime),{
    PrintMonsterType = function ()
        log("Slime")
    end
})
xlua.hotfix(CS.Monster(CS.Goblin),{
    PrintMonsterType = function ()
        log("IGoblin")
    end
})
